# 离线优先同步方案实施计划

## 方案核心：离线优先，定时同步

### 设计理念
1. **完全离线使用**：手机可以在无网络环境下正常使用
2. **本地优先存储**：所有数据先保存在手机本地
3. **待同步队列**：所有操作都添加到待同步队列
4. **服务器检测**：定时检测服务器是否可用
5. **自动同步**：当服务器可用时自动同步数据
6. **数据合并**：智能合并多设备数据，按时间戳解决冲突
7. **零成本**：无需云服务，使用现有设备

### 本地服务器实现

#### 1. 服务器功能
- 存储所有设备的笔记和相册数据
- 接收待同步队列，合并数据
- 提供最新数据下载
- 支持ping检测

#### 2. 服务器代码（server.js）
```javascript
const express = require('express');
const cors = require('cors');
const fs = require('fs');
const path = require('path');

const app = express();
const PORT = 3000;
const DATA_FILE = path.join(__dirname, 'sync-data.json');

// 中间件
app.use(cors());
app.use(express.json());

// 初始化数据文件
function initDataFile() {
  if (!fs.existsSync(DATA_FILE)) {
    const initialData = {
      notes: [],
      albums: [],
      devices: [],
      lastUpdate: new Date().toISOString()
    };
    fs.writeFileSync(DATA_FILE, JSON.stringify(initialData, null, 2));
  }
}

// 健康检查
app.get('/ping', (req, res) => {
  res.json({ status: 'ok', timestamp: new Date().toISOString() });
});

// 获取最新数据
app.get('/data', (req, res) => {
  initDataFile();
  const data = JSON.parse(fs.readFileSync(DATA_FILE, 'utf8'));
  res.json(data);
});

// 同步数据
app.post('/sync', (req, res) => {
  initDataFile();
  const { queue, deviceId } = req.body;
  
  if (!queue || !Array.isArray(queue)) {
    return res.status(400).json({ error: '无效的同步队列' });
  }
  
  // 读取现有数据
  const data = JSON.parse(fs.readFileSync(DATA_FILE, 'utf8'));
  
  // 处理同步队列
  queue.forEach(item => {
    processSyncItem(data, item, deviceId);
  });
  
  // 更新最后更新时间
  data.lastUpdate = new Date().toISOString();
  
  // 保存数据
  fs.writeFileSync(DATA_FILE, JSON.stringify(data, null, 2));
  
  res.json({ 
    success: true, 
    message: `处理了${queue.length}条同步记录`,
    lastUpdate: data.lastUpdate
  });
});

// 处理单条同步记录
function processSyncItem(data, item, deviceId) {
  const { type, dataType, data: itemData } = item;
  
  if (dataType === 'note') {
    processNoteSync(data, item, deviceId);
  } else if (dataType === 'album') {
    processAlbumSync(data, item, deviceId);
  }
}

// 处理笔记同步
function processNoteSync(data, item, deviceId) {
  const { type, data: note } = item;
  
  if (type === 'add' || type === 'update') {
    // 查找现有笔记
    const existingIndex = data.notes.findIndex(n => n.id === note.id);
    
    if (existingIndex !== -1) {
      // 现有笔记，按时间戳更新
      const existingNote = data.notes[existingIndex];
      const existingTime = new Date(existingNote.updatedAt || existingNote.createdAt);
      const newTime = new Date(note.updatedAt || note.createdAt);
      
      if (newTime > existingTime) {
        data.notes[existingIndex] = note;
      }
    } else {
      // 新笔记，直接添加
      data.notes.push(note);
    }
  } else if (type === 'delete') {
    // 删除笔记
    data.notes = data.notes.filter(n => n.id !== note.id);
  }
}

// 启动服务器
app.listen(PORT, () => {
  console.log(`服务器运行在 http://localhost:${PORT}`);
  console.log('使用 ngrok http 3000 进行内网穿透');
});
```
