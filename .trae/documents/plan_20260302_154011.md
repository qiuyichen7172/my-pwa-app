### 2.1 创建 SyncManager 类

```javascript
// SyncManager 类 - 处理数据同步
class SyncManager {
    constructor() {
        this.syncQueue = JSON.parse(localStorage.getItem('syncQueue')) || [];
        this.serverUrl = localStorage.getItem('serverUrl') || 'https://pwa.diandiandidi.vip';
        this.syncInterval = null;
        this.isSyncing = false;
    }

    // 设置服务器地址
    setServerUrl(url) {
        this.serverUrl = url;
        localStorage.setItem('serverUrl', url);
    }

    // 添加到同步队列
    addToQueue(item) {
        // 确保item有必要属性
        if (!item.id) {
            item.id = Date.now().toString();
        }
        if (!item.updatedAt) {
            item.updatedAt = new Date().toISOString();
        }
        if (!item.createdAt) {
            item.createdAt = new Date().toISOString();
        }
        
        // 检查是否已在队列中
        const existingIndex = this.syncQueue.findIndex(qItem => qItem.id === item.id);
        if (existingIndex !== -1) {
            // 更新现有项
            this.syncQueue[existingIndex] = item;
        } else {
            // 添加到队列
            this.syncQueue.push(item);
        }
        
        localStorage.setItem('syncQueue', JSON.stringify(this.syncQueue));
        console.log('[SyncManager] 添加到同步队列:', item.id);
        
        // 尝试立即同步
        this.trySync();
    }

    // 尝试同步（带防抖）
    trySync() {
        if (!navigator.onLine) return;
        if (this.isSyncing) return;
        
        this.isSyncing = true;
        setTimeout(() => {
            this.syncToServer();
        }, 1000);
    }

    // 同步到服务器
    async syncToServer() {
        if (this.syncQueue.length === 0) {
            this.updateSyncStatus(true, '已同步');
            this.isSyncing = false;
            return;
        }
        
        this.updateSyncStatus(false, '同步中...');
        console.log('[SyncManager] 开始同步到服务器，队列长度:', this.syncQueue.length);
        
        try {
            const response = await fetch(`${this.serverUrl}/notes.json`, {
                method: 'PUT',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(this.syncQueue),
            });
            
            if (response.ok) {
                // 同步成功，清空队列
                this.syncQueue = [];
                localStorage.setItem('syncQueue', JSON.stringify(this.syncQueue));
                this.updateSyncStatus(true, '已同步');
                console.log('[SyncManager] 同步到服务器成功');
            } else {
                throw new Error('同步失败');
            }
        } catch (error) {
            console.error('[SyncManager] 同步到服务器失败:', error);
            this.updateSyncStatus(true, '同步失败');
        } finally {
            this.isSyncing = false;
        }
    }

    // 从服务器拉取数据
    async pullFromServer() {
        console.log('[SyncManager] 从服务器拉取数据');
        
        try {
            const response = await fetch(`${this.serverUrl}/notes.json`);
            if (response.ok) {
                const remoteNotes = await response.json();
                const localNotes = JSON.parse(localStorage.getItem('notes')) || [];
                
                // 合并数据
                const mergedNotes = this.mergeData(localNotes, remoteNotes);
                localStorage.setItem('notes', JSON.stringify(mergedNotes));
                
                // 更新UI
                renderNotes();
                console.log('[SyncManager] 从服务器拉取数据成功，合并了', mergedNotes.length, '条笔记');
            }
        } catch (error) {
            console.error('[SyncManager] 从服务器拉取数据失败:', error);
        }
    }

    // 合并数据（时间戳优先）
    mergeData(local, remote) {
        console.log('[SyncManager] 合并数据，本地:', local.length, '条，远程:', remote.length, '条');
        
        const merged = {};
        
        // 添加本地数据
        local.forEach(note => {
            merged[note.id] = note;
        });
        
        // 添加远程数据（时间戳更新的优先）
        remote.forEach(note => {
            if (!merged[note.id] || new Date(note.updatedAt) > new Date(merged[note.id].updatedAt)) {
                merged[note.id] = note;
            }
        });
        
        // 转换为数组并按时间戳排序
        const result = Object.values(merged).sort((a, b) => {
            return new Date(b.updatedAt) - new Date(a.updatedAt);
        });
        
        console.log('[SyncManager] 合并完成，共', result.length, '条笔记');
        return result;
    }

    // 更新同步状态
    updateSyncStatus(isSynced, text) {
        const syncIcon = document.getElementById('sync-icon');
        const syncText = document.getElementById('sync-text');
        
        if (syncIcon && syncText) {
            if (isSynced) {
                syncIcon.textContent = '✅';
            } else {
                syncIcon.textContent = '🔄';
            }
            syncText.textContent = text;
        }
    }

    // 初始化同步
    init() {
        // 设置网络监听
        window.addEventListener('online', () => {
            console.log('[SyncManager] 网络已连接，开始同步');
            this.syncToServer();
            this.pullFromServer();
        });
        
        // 设置定期同步
        this.syncInterval = setInterval(() => {
            if (navigator.onLine) {
                this.syncToServer();
                this.pullFromServer();
            }
        }, 60000); // 每分钟同步一次
        
        // 初始同步
        if (navigator.onLine) {
            setTimeout(() => {
                this.pullFromServer();
            }, 1000);
        }
    }

    // 上传图片
    async uploadImage(base64Image) {
        try {
            const response = await fetch(`${this.serverUrl}/upload`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({ image: base64Image }),
            });
            
            if (response.ok) {
                const result = await response.json();
                return result.url;
            } else {
                throw new Error('图片上传失败');
            }
        } catch (error) {
            console.error('[SyncManager] 图片上传失败:', error);
            throw error;
        }
    }
}

// 初始化 SyncManager
const syncManager = new SyncManager();
```

// 处理插入媒体
function handleInsertMedia(file, type) {
    if (!file) return;
    
    // 检查文件大小
    if (file.size > 10 * 1024 * 1024) { // 10MB限制
        alert('图片大小不能超过10MB！');
        return;
    }
    
    const reader = new FileReader();
    reader.onload = async (e) => {
        const content = e.target.result;
        const editor = document.getElementById('note-content');
        
        try {
            // 压缩图片
            const compressedDataUrl = await compressImage(content);
            
            // 上传图片到服务器
            const imageUrl = await syncManager.uploadImage(compressedDataUrl);
            
            // 在光标位置插入媒体
            const selection = window.getSelection();
            const range = selection.getRangeAt(0);
            
            const img = document.createElement('img');
            img.src = imageUrl;
            img.alt = file.name;
            // 添加样式类，确保图片大小合适
            img.style.maxWidth = '100%';
            img.style.maxHeight = '400px';
            img.style.height = 'auto';
            img.style.objectFit = 'contain';
            img.style.margin = '1rem auto';
            img.style.display = 'block';
            range.insertNode(img);
            
            // 移动光标到媒体后面
            range.setStartAfter(img);
            range.collapse(true);
            selection.removeAllRanges();
            selection.addRange(range);
            
            // 触发input事件
            editor.dispatchEvent(new Event('input', { bubbles: true }));
        } catch (error) {
            console.error('[handleInsertMedia] 处理媒体失败:', error);
            alert('处理媒体失败，请重试！');
        }
    };
    reader.readAsDataURL(file);
}

// 压缩图片
function compressImage(dataUrl, quality = 0.7) {
    return new Promise((resolve, reject) => {
        const img = new Image();
        img.onload = () => {
            const canvas = document.createElement('canvas');
            
            // 计算压缩后的尺寸
            let width = img.width;
            let height = img.height;
            const maxDimension = 1200;
            
            if (width > maxDimension || height > maxDimension) {
                if (width > height) {
                    height = Math.round(height * (maxDimension / width));
                    width = maxDimension;
                } else {
                    width = Math.round(width * (maxDimension / height));
                    height = maxDimension;
                }
            }
            
            canvas.width = width;
            canvas.height = height;
            
            const ctx = canvas.getContext('2d');
            ctx.drawImage(img, 0, 0, width, height);
            
            // 转换为base64
            const compressedDataUrl = canvas.toDataURL('image/jpeg', quality);
            resolve(compressedDataUrl);
        };
        img.onerror = reject;
        img.src = dataUrl;
    });
}

## 五、service-worker.js 配置

```javascript
// 缓存名称
const CACHE_NAME = 'couple-notes-v3';
const urlsToCache = [
  '.',
  'index.html',
  'css/style.css',
  'js/script.js',
  'manifest.json',
  'images/icon-192x192.png',
  'images/icon-512x512.png'
];

// 安装Service Worker
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then((cache) => {
        console.log('Cache opened');
        return cache.addAll(urlsToCache);
      })
  );
  self.skipWaiting(); // 立即激活新的Service Worker
});

// 激活Service Worker
self.addEventListener('activate', (event) => {
  const cacheWhitelist = [CACHE_NAME];
  event.waitUntil(
    caches.keys().then((cacheNames) => {
      return Promise.all(
        cacheNames.map((cacheName) => {
          if (cacheWhitelist.indexOf(cacheName) === -1) {
            console.log('Deleting old cache:', cacheName);
            return caches.delete(cacheName);
          }
        })
      );
    }).then(() => {
      return self.clients.claim(); // 立即控制所有客户端
    })
  );
});

// 拦截网络请求
self.addEventListener('fetch', (event) => {
  // 对于API请求，优先从网络获取
  if (event.request.url.includes('/notes.json') || event.request.url.includes('/upload')) {
    event.respondWith(
      fetch(event.request)
        .then((response) => {
          return response;
        })
        .catch(() => {
          // 网络请求失败时，返回离线状态
          return new Response(JSON.stringify({ error: 'offline' }), {
            headers: { 'Content-Type': 'application/json' }
          });
        })
    );
  } else {
    // 对于静态资源，使用缓存优先策略
    event.respondWith(
      caches.match(event.request)
        .then((response) => {
          // 缓存命中，返回缓存内容
          if (response) {
            return response;
          }
          // 缓存未命中，发起网络请求
          return fetch(event.request)
            .then((response) => {
              // 检查响应是否有效
              if (!response || response.status !== 200 || response.type !== 'basic') {
                return response;
              }
              
              // 克隆响应，一份用于返回，一份用于缓存
              const responseToCache = response.clone();
              caches.open(CACHE_NAME)
                .then((cache) => {
                  cache.put(event.request, responseToCache);
                });
              
              return response;
            });
        })
    );
  }
});
```
